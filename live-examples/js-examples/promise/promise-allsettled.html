const promise1 = Promise.resolve(3);
const promise2 = 42;
const promise3 = new Promise(function(resolve, reject) {
  setTimeout(reject, 100, 'foo');
});
const promises = [promise1, promise2, promise3];

Promise.allSettled(promises).then(function(results) {
  console.log(results);
});
